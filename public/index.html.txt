    <script>
        const statusDiv   = document.getElementById('connection-status');
        const feedDiv     = document.getElementById('feed');
        const placeholder = document.getElementById('placeholder');

        // ── SESSION STORAGE: restore key across page switches ─────────────────
        let activeApiKey = sessionStorage.getItem('apiKey') || '';

        function bootSocket(apiKey) {
            const socket = io({ auth: { token: apiKey } });

            socket.on('connect', () => {
                // Save key so other pages can read it too
                sessionStorage.setItem('apiKey', apiKey);
                activeApiKey = apiKey;

                document.getElementById('login').style.display = 'none';
                document.getElementById('dashboard').style.display = 'block';
                statusDiv.textContent = "🟢 WebSocket Connected — UI Live";
                statusDiv.className = "status connected";
            });

            socket.on('connect_error', (err) => {
                // Bad key — wipe it so user sees login again on refresh
                sessionStorage.removeItem('apiKey');
                document.getElementById('loginError').textContent = err.message;
            });

            socket.on('disconnect', () => {
                statusDiv.textContent = "🔴 WebSocket Disconnected";
                statusDiv.className = "status disconnected";
            });

            socket.on('live-ticker-update', (data) => {
                if (placeholder) placeholder.remove();

                const item = document.createElement('div');
                item.className = 'tick-item';

                const sideClass  = data.side.toUpperCase() === 'BUY' ? 'side-buy' : 'side-sell';
                const localTime  = new Date(data.timestamp).toLocaleTimeString();

                item.innerHTML = `
                    <div>
                        <span class="ticker-name">${data.ticker}</span>
                        <span class="time">(${localTime})</span>
                    </div>
                    <div>
                        <span class="${sideClass}">${data.side.toUpperCase()}</span>
                        <span class="price" style="margin-left: 20px;">₹${Number(data.price).toFixed(2)}</span>
                    </div>
                `;
                feedDiv.insertBefore(item, feedDiv.firstChild);
            });
        }

        // ── AUTO-LOGIN if key already in sessionStorage ───────────────────────
        document.addEventListener('DOMContentLoaded', () => {
            if (activeApiKey) {
                bootSocket(activeApiKey);
            }
        });

        // ── MANUAL LOGIN ──────────────────────────────────────────────────────
        document.getElementById('connectBtn').addEventListener('click', () => {
            const apiKey = document.getElementById('apiKeyInput').value.trim();
            if (!apiKey) return;
            bootSocket(apiKey);
        });

        // ── ORDER SUBMISSION ──────────────────────────────────────────────────
        document.getElementById('submitOrderBtn').addEventListener('click', async () => {
            const ticker    = document.getElementById('orderTicker').value;
            const price     = parseFloat(document.getElementById('orderPrice').value);
            const quantity  = parseInt(document.getElementById('orderQuantity').value);
            const side      = document.getElementById('orderSide').value;
            const statusMsg = document.getElementById('orderStatusMsg');

            statusMsg.style.color   = '#64748b';
            statusMsg.textContent   = "Submitting order...";

            try {
                const response = await fetch('/api/orders', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'x-api-key': activeApiKey
                    },
                    body: JSON.stringify({ ticker, price, quantity, side })
                });

                if (response.ok) {
                    statusMsg.style.color   = '#4ade80';
                    statusMsg.textContent   = "Order accepted by Ingestion Layer!";
                    setTimeout(() => { statusMsg.textContent = ""; }, 3000);
                } else {
                    const errText = await response.text();
                    statusMsg.style.color   = "#f87171";
                    statusMsg.textContent   = `Rejected: ${response.status} (${errText})`;
                }
            } catch (err) {
                statusMsg.style.color   = '#f87171';
                statusMsg.textContent   = `Error: ${err.message}`;
            }
        });
    </script>
