import { Kafka } from 'kafkajs';
import dotenv from 'dotenv';

dotenv.config();

const kafka = new Kafka({
    clientId : 'market-simulator',
    brokers : [process.env.KAFKA_BROKERS || 'localhost:9092'] 
});

const producer = kafka.producer();

//Track memory state for the price walk
const MARKET_STATE = {
    INFY : { lastPrice : 1500.00 },
    RELIANCE : {lastPrice : 2400.00} ,
    TCS : {lastPrice : 3200.00},
    HDFCBANK : {lastPrice : 4000.00}
};

const TICKERS = Object.keys(MARKET_STATE);
const SIM_USERS = ['SIM_USER_VARUN', 'SIM_USER_XYZ'];
const USER_IDS = ['6a1b3ecf3fc9d5361bcf263d', '6a1dd7469a53d1d1623887de'];

//Generate a price walk order payload
const generatePriceWalkOrder = ( tickCount) => {
    const ticker = TICKERS[Math.floor(Math.random() * TICKERS.length)];
    const state = MARKET_STATE[ticker];

    //1. PRICE WALK MECHANISM : MAX +/- 0.5% shift from the last generated price

    const maxChangePercent = 0.005;
    const changePercent = (Math.random() * 2 - 1 ) * maxChangePercent;
    let targetPrice = state.lastPrice * (1 + changePercent);

    //2. MISMATCH AND SPREAD DYNAMICS
    //We intentionally tweak bids/asks slightly so they don;t always instantly match.

    const side = Math.random() > 0.5? 'BUY' : 'SELL';
    if(side === 'BUY'){
        //Buyers sometimes try to lowball slightly below the true current ticker value 
        targetPrice -= (Math.random() * 0.002) * targetPrice;
    }
    else{
        //Sellers sometimes try to ask for slightly more 
        targetPrice += (Math.random() * 0.002) * targetPrice;
    }

    const finalPrice = parseFloat(targetPrice.toFixed(2));

    //Sace this baseline back to memory so the next tick walks from here

    state.lastPrice = finalPrice;

    //3. VARIABLE QUANTITY MECHANISM
    const quantity = Math.floor(Math.random() * 100 ) + 1;

    //4. USER ALTERNATION MECHANISM
    const userId = USER_IDS[tickCount % USER_IDS.length];

    return{
        orderId : `sim_${Math.random().toString(36).substring(2,9)}`,
        userId,
        ticker,
        side,
        price : finalPrice,
        quantity : quantity,
        timestamp :  new Date().toISOString()
    };
    
};

const StartSimulation = async () => {
    try{
        await producer.connect();
        console.log('Highfrequency Price Walk Simulator Initialized');
        console.log( ` Tracking baseline trends for ${TICKERS.join  (', ')} `);

        let tickCount = 0;
        //Fire every 500ms to put realistic concurrency pressure on the matching loops
        setInterval( async () => {
            const orderPayload = generatePriceWalkOrder(tickCount++);

            try{
                await producer.send({
                    topic : 'order-requests',
                    messages : [
                        {
                            key: orderPayload.ticker,
                            value: JSON.stringify(orderPayload)
                        }
                    ]
                });

                //Printclear visiblity data to the terminal
                const userTag = orderPayload.userId === USER_IDS[0]? 'VARUN' : 'XYZ_T';       
                console.log(`[${userTag}] -> ${orderPayload.side} ${orderPayload.ticker} | Qty : ${orderPayload.quantity} @ rs${orderPayload.price}`);
            }
            catch(err){
                console.error("Failed to send price walk order" , err.message);
            }
        },500);
    
    } catch(error){
        console.error("Critical Error inside  Simulator Engine" , error);
        process.exit(1);
    }
};

StartSimulation();
