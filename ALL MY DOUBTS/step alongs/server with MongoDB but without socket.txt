import express from 'express';
import { Kafka } from 'kafkajs';
import mongoose from 'mongoose';
import dotenv from 'dotenv';
import connectDB from './db.js';
import { Trade } from './Trade.js';

//Load environment variables from .env file
dotenv.config();

const app = express();
const PORT = 5000;

app.use(express.json());


// 1 . INITIALIZE KAFKA CLIENT 
// connects to the external port we exposed in our docker - compose file 


const kafka = new Kafka({
    clientId : 'financial-app',     //name of the backend server that is connecting to kafka cluster 
    brokers : ['localhost:9092'] // this is the default port for kafka broker
});

const producer = kafka.producer();
const consumer = kafka.consumer({ groupId : 'analytics-group '});

//2 . START SERVICES (MONGO -> KAFKA -> EXPRESS)
const startServices = async() => {
    try{

        // A. Connect to MongoDB Atlas first, since our Kafka consumer will need to read/write from the database in a moment
        console.log("Connecting to MongoDB Atlas..");
        await mongoose.connect(process.env.MONGODB_URI);
        console.log("MongoDB Connected Successfully");


        await producer.connect();  //to wait for the connection to establish before moving to the next line of code 
        await consumer.connect(); //else we might get a consumer not connected error if we try to subscribe before the connection is ready
        console.log("Kafka Producer and Consumer Connected successfully"); 

        //subscribe our consumer to the 'trade - ticks' topic
        await consumer.subscribe({ topic : 'trade-ticks' , fromBeginning : true});

        //start the  background Consumer loop
        await consumer.run({
            eachMessage : async({ topic : partition , message}) => {
                // Parse the binary buffer back into usable JSON string data
                const tick = JSON.parse(message.value.toString());
                console.log(`\n Real Kafka Consumer : [CALCULATING METRICS]`);
                console.log(`Ticker : ${tick.ticker} | Price rs${tick.price} | Side ${tick.side}`);

                try{
                    //C. Save the live trade tick into MongoDB Atlas
                    const newTrade = new Trade({
                        ticker : tick.ticker,
                        price : tick.price,
                        side : tick.side
                    });

                    await newTrade.save();
                    console.log(`Saved tick for ${tick.ticker} securely to MongoDB Atlas`);
                
                }
                catch(dbError){
                    console.error("Failed to save tick to MongDB" , dbError.message);
                }
            } ,
        }); 
//Start Express server only after Kafka is completely ready
        app.listen(PORT , () => {
            console.log(`INGESTION SERVER IS LIVE ON http://localhost:${PORT}`);
        });

    } catch(error){
    console.error("Critical Service Initialization Error" , error);
    process.exit(1);
    }
};

// 3 . THE INGESTION ROUTE (The producer)

app.post('/api/ticker' , async(req , res) => {
    try {
        const tradeTick = req.body;

        //send the payload straight onto the real Kafka conveyor belt
        await producer.send({  //implictly converts the string into a binary buffer and sends it to the broker
            topic : 'trade-ticks', 
            messages : [
                {value : JSON.stringify(tradeTick)}
            ],
        });
        //Instantly tell Thunder Client we queued the data safely 
        return res.status(202).json({
            success : true ,
            message : "Trade tick successfully broadcasted to Kafka"
        });
    }
    catch(error){
        console.effect("Kafka Producer Error" , error);
        return res.status(500).json({success : false , error : error.message});

    }
});

// Fire up the start sequence
startServices().catch(console.error);
