import express from 'express';
import { Kafka } from 'kafkajs';
import mongoose from 'mongoose';
import dotenv from 'dotenv';
import http from 'http'; //Native Node HTTP module 
import { Server } from 'socket.io';
import { User , HistoricalTrade } from './model.js';
import connectDB from './db.js';
import { createClient } from 'redis';

//Load environment variables from .env file
dotenv.config();

const app = express();
const PORT = 5000;

//Create the unified HTTP server wrapper around Express
const server = http.createServer(app);

//Initialize Socket.io server layer on top of our HTTP server 
const io = new Server( server , {
    cors : { origin : "*"}  // Protects and allows frontends to listen securely 
});

io.use(async (socket , next) => {
    const apiKey = socket.handshake.auth.token;
    if(!apiKey) return next(new Error('API key missing'));

    const user = await User.findOne({ apiKey });
    if(!user ) return next(new Error('Invalid API key'));

    socket.userId = user._id.toString();
    socket.join(socket.userId);
    next();
});

app.use(express.json());

//Tell Express to automatically serve files inside the 'public' folder
app.use(express.static('public'));

// 1 . INITIALIZE KAFKA CLIENT 
const kafka = new Kafka({
    clientId : 'financial-app',     
    brokers : ['localhost:9092'] 
});

const producer = kafka.producer();
const consumer = kafka.consumer({ groupId : 'analytics-group'});

const redisClient  = createClient({
    url : process.env.REDIS_URL || 'redis://localhost:6379'
});

redisClient.on('error' , (err) => console.error('redis Client Error',err));

//2 . START SERVICES (MONGO -> KAFKA -> EXPRESS)
const startServices = async() => {
    try{
        await connectDB(); 
        await redisClient.connect();
        console.log('Redis Connected Successfully');
        await producer.connect();  
        await consumer.connect(); 
        console.log("Kafka Producer and Consumer Connected successfully"); 

        await consumer.subscribe({ topic : 'order-requests' , fromBeginning : true});

        //start the background Consumer loop (with stateful matching engine)
        await consumer.run({
            eachMessage : async({topic , partition , message}) => {
                const incomingOrder = JSON.parse(message.value.toString());
                const { orderId , userId , ticker , side , price , quantity }  = incomingOrder;
                
                console.log(`\n📦 Engine Received Order Intent : ${side} ${ticker} | Qty ${quantity} @ ${price}`);

                const buyBookKey = `orders:${ticker}:BUY`;
                const sellBookKey = `orders:${ticker}:SELL`;
                let remainingQty = quantity;

                try {
                    if(side === 'BUY'){
                        //1 . MATCHING LOOP FOR BUY ORDERS
                        while (remainingQty > 0){
                            const bestSells = await redisClient.zRangeWithScores(sellBookKey , 0 , 0);
                            if(bestSells.length === 0) break; 

                            const cheapestSellOrder = JSON.parse(bestSells[0].value);
                            const cheapestSellPrice = bestSells[0].score;

                            if(price >= cheapestSellPrice){
                                const matchQty = Math.min(remainingQty , cheapestSellOrder.quantity);
                                console.log(`⚡ EXECUTION : Matched ${matchQty} shares of ${ticker} @ ${cheapestSellPrice}`);

                                remainingQty -= matchQty;
                                cheapestSellOrder.quantity -= matchQty;

                                // PERSISTS THE MATCHED TRANSACTION TO MONGO AS A REAL HISTORICAL TRADE
                                const buyTrade  = new HistoricalTrade({
                                    userId ,
                                    ticker , price: cheapestSellPrice , quantity : matchQty , side : 'BUY'
                                });

                                const sellTrade = new HistoricalTrade({
                                    userId : cheapestSellOrder.userId,
                                    ticker , price : cheapestSellPrice , quantity : matchQty , side : 'SELL'
                                });

                                await Promise.all([buyTrade.save() , sellTrade.save()]);

                                //STREAM REAL-TIME MATCH NOTIFICATIONS TO BOTH USERS SEPARATELY VIA SOCKET ROOMS
                                io.to(userId).emit('live-ticker-update', {ticker , price : cheapestSellPrice , side : 'BUY' , timestamp : new Date()});
                                io.to(cheapestSellOrder.userId).emit('live-ticker-update' , {ticker , price : cheapestSellPrice , side : 'SELL' , timestamp : new Date()});

                                await redisClient.zRem(sellBookKey , bestSells[0].value);
                                if(cheapestSellOrder.quantity > 0){
                                    await redisClient.zAdd(sellBookKey , {score : cheapestSellPrice , value : JSON.stringify(cheapestSellOrder)});
                                } 
                            } else {
                                break;
                            }      
                        }
                        
                        if(remainingQty > 0){
                            const partialOrder = { orderId , userId , ticker , side , price , quantity : remainingQty};
                            await redisClient.zAdd(buyBookKey , { score : price , value : JSON.stringify(partialOrder)});
                        }
                    }
                    else if( side === 'SELL'){
                        //2. MATCHING LOOP FOR SELL ORDERS
                        while(remainingQty > 0){
                            const bestBuys = await redisClient.zRangeWithScores(buyBookKey , -1, -1);
                            if(bestBuys.length === 0 ) break;

                            const highestBuyOrder = JSON.parse(bestBuys[0].value);
                            const highestBuyPrice = bestBuys[0].score;
                            
                            if(price <= highestBuyPrice){
                                const matchQty = Math.min(remainingQty , highestBuyOrder.quantity);
                                console.log(`⚡ EXECUTION : Matched ${matchQty} shares of ${ticker} @ ${highestBuyPrice}`);

                                remainingQty -= matchQty;
                                highestBuyOrder.quantity -= matchQty;

                                const sellTrade = new HistoricalTrade({
                                    userId,
                                    ticker, price : highestBuyPrice , quantity : matchQty , side : 'SELL'
                                });
                                const buyTrade = new HistoricalTrade({
                                    userId : highestBuyOrder.userId,
                                    ticker , price : highestBuyPrice , quantity : matchQty , side : 'BUY'
                                });

                                await Promise.all([sellTrade.save() , buyTrade.save()]);

                                io.to(userId).emit('live-ticker-update' , {ticker , price : highestBuyPrice , side : 'SELL' , timestamp : new Date()});
                                io.to(highestBuyOrder.userId).emit('live-ticker-update' , {ticker , price : highestBuyPrice , side : 'BUY' , timestamp : new Date()});
                                
                                await redisClient.zRem(buyBookKey , bestBuys[0].value);
                                if(highestBuyOrder.quantity > 0){
                                    await redisClient.zAdd(buyBookKey , {score : highestBuyPrice , value : JSON.stringify(highestBuyOrder)});
                                }
                                else{
                                    break;
                                }
                            }
                            else {
                                break;
                            }

                            if(remainingQty > 0){
                                const partialOrder = {orderId , userId , ticker , side , price , quantity : remainingQty};
                                await redisClient.zAdd(sellBookKey , {score : price , value : JSON.stringify(partialOrder)});
                            }
                        }
                    }
                } catch(dbError){
                    console.error("Critical Engine Loop Processing Fault" , dbError.message);
                }
            },
        });

        server.listen(PORT , () => {
            console.log(`INGESTION SERVER & UI PORTAL LIVE ON http://localhost:${PORT}`);
        });

    } catch(error){
        console.error("Critical Service Initialization Error" , error);
        process.exit(1);
    }
};

// 3 . THE INGESTION ROUTE (The producer)
app.post('/api/orders' , async(req , res) => {
    try {
        const apiKey = req.headers['x-api-key'];
        if(!apiKey) return res.status(401).json({ success : false , error : "API key missing"});

        const user = await User.findOne({ apiKey });
        if(!user) return res.status(403).json({ success : false , error : "Invalid API key"});

        const { ticker , price , quantity , side } = req.body;

        //sanity guardrails----
        if(!ticker || !side || price === undefined || quantity === undefined ){
            return res.status(400).json({sucess : false , error : "Missing required fields"});
        }

        const parsedSide = side.toUpperCase();
        if(parsedSide!= 'BUY' && parsedSide!== 'SELL'){
            return res.status(400).json({sucess: false , error : "Invalid side . Must be 'BUY' or 'SELL'"});

        }

        const parsedPrice = parseFloat(price);
        const parsedQuantity = parseInt(quantity);

        //Block null , NaN , zero , negatives , or ridiculous break the engine integers
        if(isNaN(parsedPrice) || parsedPrice <=0){
            return res.status(400).json({sucess : false , error : "Invalid price"});
        }

        if(isNaN(parsedQuantity) || parsedQuantity <=0){
            return res.status(400).json({sucess : false , error : "Invalid quantity"});
        }
        //----
        const orderId = `ord_${Math.random().toString(36).substring(2,9)}`;
        
        const orderPayload ={
            orderId ,
            userId : user._id.toString(),
            ticker : ticker.toUpperCase(),
            side : side.toUpperCase(),
            price : parseFloat(price),
            quantity : parseInt(quantity),
            timestamp : new Date().toISOString()
        };

        await producer.send({  
            topic : 'order-requests', 
            messages : [
                {key : ticker.toUpperCase() , value :  JSON.stringify(orderPayload)},
            ],
        });

        return res.status(202).json({
            success : true ,
            message : "Trade tick successfully broadcasted to Kafka" , orderId 
        });
    }
    catch(error){
        console.error("Kafka Producer Error" , error);
        return res.status(500).json({success : false , error : error.message});
    }
});

startServices().catch(console.error);
